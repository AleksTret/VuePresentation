main.js
const app = new Vue({
  el: '#app',
  data: {
    message: 'Hello, Vue!',
    now: new Date(),
  },
  methods: {
    updateDate() {
      this.now = new Date();
    }
  },
  mounted(){
    setInterval(() => this.updateDate(), 1000);
  }
})
